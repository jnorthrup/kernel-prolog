710  public int exec(Prog p) {
711 
712  Term o=getArg(0);
713  Term a=getArg(1);
714  Term b=getArg(2);
715  if(!(o instanceof Const)||!(a instanceof Num)||!(b instanceof Num))
716  IO.errmes("bad arithmetic operation ("+o+"): "+a+","+b+"\nprog: "
717  +p.toString());
718  String opname=((Const)o).name();
719  double x=((Num)a).getValue();
720  double y=((Num)b).getValue();
721  double r;
722  char op=opname.charAt(0);
723  switch(op) {
724  case '+':
725  r=x+y;
726  break;
727  case '-':
728  r=x-y;
729  break;
730  case '*':
731  r=x*y;
732  break;
733  case '/':
734  r=x/y;
735  break;
736  case ':':
737  r=(int)(x/y);
738  break;
739  case '%':
740  r=x%y;
741  break;
742  case '?':
743  r=(x<y)?(-1):(x==y?0:1);
744  break; // compares!
745  case 'p':
746  r=Math.pow(x,y);
747  break;
748  case 'l':
749  r=Math.log(y)/Math.log(x);
750  break;
751  case 'r':
752  r=x*Math.random()+y;
753  break;
754  case '<':
755  r=(long)x<<(long)y;
756  break;
757  case '>':
758  r=(long)x>>(long)y;
759  break;
760 
761  default:
762  IO.errmes("bad arithmetic operation <"+op+"> on "+x+" and "+y);
763  return 0;
764  }
765  Num R=((Math.floor(r)==r))?(Num)(new Int((long)r)):(Num)(new Real(r));
766  return putArg(3,R,p);
